1.基础查询

1.查询字段

select 字段列名  from 表名;
select * from 表名;//不要使用！！

2.查询时去除重复元素

select distinct 字段列名 from 表名;

3.起别名

select 字段列名 as 别名 from 表名;
as可省略

2.条件查询

语法
select 列名 from 表名 where 条件;
条件
  1. 等于：=
  2. 某个范围之间：列名 between 范围1 and 范围2
  3. 或者||：or
  4. 在……之中：in（条件1，条件2……），满足一个即可
  5. 并且&&：and
  6. 是空：is null
  7. 不是空：is not null
模糊查询

like:%表示多个字符 _表示单个字符，第一个列名可以用*表示全部

select 列名 from 表名 where 列名 like '条件';
例如：
select * from stu where name like '马%';//姓马的人

3.排序查询

升序asc，降序desc

select 列名 from 表名 order by 列名1 排序方式1,列名2 排序方式2……;
例如：
select 列名 from stu order by math desc,english asc;
先按数学降序排，再按英语升序排

前面排序内容相同时，才用后面的排序

4.分组查询

聚合函数

将一列作为一个整体，纵向计算，聚合函数均不计算null

  1. count(列名)：统计数量（求有几列）
  2. max(列名)：最大值 min(列名)：最小值
  3. sum(列名)：求和（求这一列的和）
  4. ave(列名):求平均值（求这一列的平均值）
用法

列名可以用* 表示全部

select 聚合函数名(列名) from 表名;
分组
语法
select 分组列名，聚合函数 from 表名 where 分组前的条件 group by 分组列名 having 分组后的条件

例如

-- 1.查询男同学和女同学各自的数学平均分
select sex,avg(math) from stu group by sex;
-- 2.查询男同学和女同学各自的数学平均分，以及各自的人数
select sex,avg(math),count(*) from stu group by sex;
-- 3.查询男同学和女同学各自的数学平均分，以及人数，分数低于70不参与评分
select sex,avg(math),count(*) from stu where math>70 group by sex;
-- 4.在3的基础上，人数要大于2
select sex,avg(math),count(*) from stu where math>70 group by sex having count(*)>2;

where和having的区别：

  1. 执行时机不同，where是分组之前限定，不满足where不参与分组，having是分组之后的过滤
  2. 可判断条件不同，where不能对聚合函数进行判断，having可以

5.分页查询

语法
select 列名 from 表名 limit 起始索引,查询条数;

起始索引：开始的位置

计算公式：起始索引=（当前页码-1）*每页条数

例如

-- 每页显示三条数据，查询第2页数据
select * from stu limit 3,3;

数据库

约束

概念：限制加入表中的数据

分类

  1. 非空：保证列中所有数据不能有null，not null
  2. 唯一：保证列中所有数据各不相同，unique
  3. 主键：一行数据的唯一标识，非空且唯一，例如：ID，primary key
  4. 默认：保存数据时，未指定值就采用默认值，default
  5. 外键：让两个表建立连接，保证数据唯一，且完整，foreign key

注意：默认约束只有不给值时才默认，如果给null结果是null而不是默认

例如

create table emp(
id int PRIMARY KEY,	-- 员工id，主键，且逐渐增加
ename varchar(50) NOT NULL UNIQUE,	-- 员工姓名，非空且唯一
joindate date NOT NULL,	-- 入职日期，非空
salary double(7,2) NOT NULL,	-- 工资，非空
bonus double(7,2) DEFAULT 0	-- 奖金，如果没有默认为0
);
外键

Nacicat test21

添加外键

1.创建两个表之前，在子表添加外键

-- 在子表内部写
CONSTRAINT 外键名称 FOREIGN KEY(子表列名) REFERENCES 主表(主表列名);
例如
CONSTRAINT fk_emp_dept FOREIGN KEY(dep_id) REFERENCES dept(id);

2.创建表之后添加外键

alter table 子表名 add CONSTRAINT 外键名称 FOREIGN key(子表列名) REFERENCES 主表(主表列名);
例如
alter table emp add CONSTRAINT fk_emp_dept FOREIGN key(dep_id) REFERENCES dept(id);

3.删除外键

alter table 子表名 drop FOREIGN key 外键名称;
例如
alter table emp drop FOREIGN key fk_emp_dept;

数据库设计

表关系

一对一：用户和用户详情，用于表的拆分

一对多：部门和员工

多对多：商品和订单

多表查询

从多张表里面查数据

  1. 连接查询：
    1. 内连接：相当于查询A，B交集数据
    2. 外连接：
      1. 左外连接：相当与查询A表所有数据和交集部分数据
      2. 右外连接：相当与查询B表所有数据和交集部分数据
  2. 子查询

内连接

-- 隐式内连接
select 列名 from 表1，表2… where 条件;
例如
select * from emp,dept where emp.dep_id=dept.did;
-- 显示内连接
select 列名 from 表1 join 表2 on 条件; 
例如
select * from emp join dept on emp.dep_id=dept.did;

外连接

-- 左外连接
-- 查询emp表所有数据和对应的员工信息
select 列名 from 表1 left join 表2 on 条件;
例如
select * from emp left join dept on emp.dep_id=dept.did;

-- 右外连接
-- 查询dept表所有数据和对应的员工信息
select 列名 from 表1 right join 表2 on 条件;
例如
select * from emp right join dept on emp.dep_id=dept.did;

子查询

嵌套查询

-- 查询财务部和市场部所有员工信息
-- 单行单列：一个条件
select 列名 from 表名 where 条件 ()=(子查询);
-- 例如
select did from dept where dname='财务部';
select * from emp where dep_id=(select did from dept where dname='财务部');

-- 多行单列，多个条件
select 列名 from 表名 where 条件 ()in(子查询);
-- 例如
select did from dept where dname='财务部' or dname='市场部';
select * from emp where dep_id=3 or dep_id=2;
-- 合并
select * from emp where dep_id in(select did from dept where dname='财务部' or dname='市场部');

-- 多行多列,需要给子查询起别名,不然太长了
select 列名 from (子查询) where 条件;
-- 例如
-- 查询入职日期是'2011-11-11'之后的员工信息和部门信息
select * from emp where join_date>'2011-11-11';
select * from emp,dept where emp.dep_id=dept.did;
-- 合并
select * from (select * from emp where join_date>'2011-11-11')t1,dept where t1.dep_id=dept.did;

commit;

DriverManager

Connection conn = DriverManager.getConnection(url,user,password); //获取连接

参数

  1. url：
    1. 语法：jdbc:mysql://IP地址:端口号/数据库名称 //端口号默认3306
    2. 示例：jdbc:mysql://localhost:3306/db1
    3. 本机可简写为：jdbc:mysql:///db1
  2. user：用户名
  3. password：密码
